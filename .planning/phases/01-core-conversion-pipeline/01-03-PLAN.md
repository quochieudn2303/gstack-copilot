   ```typescript
   import type { CopilotFrontmatter } from '../schemas/copilot.js';
   
   export interface OutputOptions {
     // Future: allow customizing output format
   }
   
   export function generateOutput(
     frontmatter: CopilotFrontmatter,
     content: string,
     options?: OutputOptions
   ): string {
     // Generate YAML frontmatter with controlled formatting
     // Use yaml package for proper escaping and quoting
     const yamlStr = stringify(frontmatter, {
       lineWidth: 0,        // Don't wrap lines
       defaultStringType: 'QUOTE_DOUBLE' as const,  // Quote strings for safety
     });
     
     // Assemble final output
     // Ensure content starts with newline for proper separation
     const normalizedContent = content.startsWith('\n') ? content : `\n${content}`;
     
     return `---\n${yamlStr}---\n${normalizedContent}`;
   }
   ```

   ```typescript
   import { readFile } from 'fs/promises';
   import { parseSkill, ConversionError, type ParsedSkill } from './parse.js';
   import { transformFrontmatter } from './frontmatter.js';
   import { transformContent } from './content.js';
   import { generateOutput } from './output.js';
   import type { CopilotFrontmatter } from '../schemas/copilot.js';
   
   export { ConversionError } from './parse.js';
   
   export interface ConversionResult {
     frontmatter: CopilotFrontmatter;
     content: string;
     output: string;
     source: ParsedSkill;
   }
   
   /**
    * Convert a gstack SKILL.md to Copilot format.
    * 
    * Pipeline: Parse → Transform Frontmatter → Transform Content → Generate Output
    * 
    * @param source - Raw SKILL.md content
    * @param filepath - Source file path (for error messages)
    * @returns Conversion result with output string
    * @throws ConversionError on validation failures
    */
   export function convertSkill(source: string, filepath: string): ConversionResult {
     // Stage 1: Parse
     const parsed = parseSkill(source, filepath);
     
     // Stage 2: Transform frontmatter
     const frontmatter = transformFrontmatter(parsed.data);
     
     // Stage 3: Transform content
     const content = transformContent(parsed.content);
     
     // Stage 4: Generate output
     const output = generateOutput(frontmatter, content);
     
     return { frontmatter, content, output, source: parsed };
   }
   
   /**
    * Convert a SKILL.md file from disk.
    */
   export async function convertSkillFile(filepath: string): Promise<ConversionResult> {
     const source = await readFile(filepath, 'utf-8');
     return convertSkill(source, filepath);
   }
   ```

   ```typescript
   import { Command } from 'commander';
   import { readFile, writeFile, mkdir } from 'fs/promises';
   import { dirname, basename, join } from 'path';
   import pc from 'picocolors';
   import { convertSkillFile, ConversionError } from '../pipeline/index.js';
   
   export const convertCommand = new Command('convert')
     .description('Convert a gstack SKILL.md to Copilot format')
     .argument('<input>', 'Path to gstack SKILL.md file')
     .option('-o, --output <path>', 'Output path (default: stdout or ./output/)')
     .option('-d, --output-dir <dir>', 'Output directory for batch conversion')
     .option('--dry-run', 'Show what would be converted without writing')
     .action(async (input: string, options: { output?: string; outputDir?: string; dryRun?: boolean }) => {
       try {
         console.log(pc.blue(`Converting: ${input}`));
         
         const result = await convertSkillFile(input);
         
         if (options.dryRun) {
           console.log(pc.yellow('\n--- DRY RUN OUTPUT ---'));
           console.log(result.output);
           console.log(pc.yellow('--- END DRY RUN ---\n'));
           return;
         }
         
         if (options.output) {
           await mkdir(dirname(options.output), { recursive: true });
           await writeFile(options.output, result.output, 'utf-8');
           console.log(pc.green(`✓ Written to: ${options.output}`));
         } else if (options.outputDir) {
           const outPath = join(options.outputDir, basename(input));
           await mkdir(options.outputDir, { recursive: true });
           await writeFile(outPath, result.output, 'utf-8');
           console.log(pc.green(`✓ Written to: ${outPath}`));
         } else {
           // Output to stdout
           console.log(result.output);
         }
         
         console.log(pc.green(`✓ Converted: ${result.frontmatter.name}`));
         
       } catch (error) {
         if (error instanceof ConversionError) {
           console.error(pc.red(`✗ Conversion failed: ${error.message}`));
           if (error.filepath) {
             console.error(pc.red(`  File: ${error.filepath}`));
           }
           process.exit(1);
         }
         throw error;
       }
     });
   ```
