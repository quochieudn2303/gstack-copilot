1. src/pipeline/parse.ts:

   ```typescript
   import matter from 'gray-matter';
   import { GstackFrontmatter } from '../schemas/gstack.js';
   
   export class ConversionError extends Error {
     constructor(message: string, public filepath?: string, public line?: number) {
       super(message);
       this.name = 'ConversionError';
     }
   }
   
   export interface ParsedSkill {
     data: GstackFrontmatter;  // Validated frontmatter
     content: string;          // Markdown body
     filepath: string;         // Source file path
   }
   
   export function parseSkill(source: string, filepath: string): ParsedSkill {
     // Use gray-matter to extract frontmatter
     const { data, content } = matter(source);
     
     // Validate frontmatter against schema (strict mode per CONTEXT.md)
     const result = GstackFrontmatter.safeParse(data);
     if (!result.success) {
       const errors = result.error.errors.map(e => 
         `  - ${e.path.join('.')}: ${e.message}`
       ).join('\n');
       throw new ConversionError(
         `Invalid gstack frontmatter in ${filepath}:\n${errors}`,
         filepath
       );
     }
     
     return { data: result.data, content, filepath };
   }
   ```

   ```typescript
     'Agent': 'Task',

   };
   
   export function mapTool(tool: string): string {
   ```

2. src/pipeline/frontmatter.ts:
   ```typescript
   import type { GstackFrontmatter } from '../schemas/gstack.js';
   import type { CopilotFrontmatter } from '../schemas/copilot.js';
   import { CopilotFrontmatter as CopilotSchema } from '../schemas/copilot.js';
   import { mapTools } from '../mappings/tools.js';
   
   export function transformFrontmatter(source: GstackFrontmatter): CopilotFrontmatter {
     // Flatten multiline description
     let description = source.description.replace(/\n+/g, ' ').trim();
     
     // Truncate if > 200 chars
     if (description.length > 200) {
       description = description.slice(0, 197) + '...';
     }
     
     const result = {
       name: `gstack-${source.name}`,
       description,
       'argument-hint': '[options]',  // Default, can be overridden
       'allowed-tools': mapTools(source['allowed-tools']).join(', '),
     };
     
     // Validate output (fail fast on any issues)
     return CopilotSchema.parse(result);
   }
   ```

1. src/mappings/paths.ts - Path transformation registry:
   ```typescript
   // Path mappings from research
   export interface PathMapping {
     pattern: RegExp;
     replacement: string;
     description: string;
   }
   
   export const PATH_MAPPINGS: PathMapping[] = [
     // Skill references - must come before generic claude path
     {
       pattern: /~\/\.claude\/skills\/gstack\/([a-z-]+)\//g,
       replacement: '~/.copilot/skills/gstack-$1/',
       description: 'Cross-skill references'
     },
     // Generic gstack paths
     {
       pattern: /~\/\.claude\/skills\/gstack\//g,
       replacement: '~/.copilot/skills/gstack-copilot/',
       description: 'gstack skill root'
     },
     {
       pattern: /~\/\.gstack\//g,
       replacement: '$env:USERPROFILE\\.gstack-copilot\\',
       description: 'gstack config directory'
     },
     // Environment variables
     {
       pattern: /\$GSTACK_BIN/g,
       replacement: '$env:GSTACK_COPILOT_BIN',
       description: 'gstack bin directory'
     },
     {
       pattern: /\$GSTACK_ROOT/g,
       replacement: '$env:GSTACK_COPILOT_ROOT',
       description: 'gstack root directory'
     },
   ];
   
   export function transformPaths(content: string): string {
     let result = content;
     for (const mapping of PATH_MAPPINGS) {
       result = result.replace(mapping.pattern, mapping.replacement);
     }
     return result;
   }
   ```

2. src/pipeline/content.ts:
   ```typescript
   import { transformPaths } from '../mappings/paths.js';
   
   // Copilot initialization block to replace gstack preamble
   const COPILOT_PREAMBLE = `## Initialization
   
   \`\`\`powershell
   # gstack-copilot initialization
   $GSTACK_COPILOT_ROOT = "$env:USERPROFILE\\.gstack-copilot"
   $GSTACK_COPILOT_BIN = "$GSTACK_COPILOT_ROOT\\bin"
   \`\`\``;
   
   export function transformContent(content: string): string {
     let result = content;
     
     // Replace gstack preamble section
     // Pattern: ## Preamble (run first) followed by bash code block
     const preamblePattern = /## Preamble \(run first\)\s*\n\n```bash[\s\S]*?```/;
     result = result.replace(preamblePattern, COPILOT_PREAMBLE);
     
     // Transform all paths
     result = transformPaths(result);
     
     return result;
   }
   ```
